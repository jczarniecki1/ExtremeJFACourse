security = require '../utilities/security' User = require 'mongoose' .model 'User' exports.getUsers = (req, res) -> User.find({}).exec (err, collection) -> res.send collection exports.createUser = (req, res, next) -> userData = req.body userData.username = userData.username userData.salt = security.createSalt() userData.hashed_pwd = security.hashPwd userData.salt, userData.password User.create userData, (err, user) -> if err? if err.toString().contains 'E11000' err = new Error 'Duplicate Username' res.status 400 res.send reason: err.toString() else req.logIn user, (err) -> res.send user.getData() exports.updateUser = (req, res, next) -> userUpdates = req.body req.user.username = userUpdates.username req.user.firstName = userUpdates.firstName req.user.lastName = userUpdates.lastName newPassword = userUpdates.password if newPassword?.length > 0 req.user.salt = security.createSalt() req.user.hashed_pwd = security.hashPwd req.user.salt, newPassword
